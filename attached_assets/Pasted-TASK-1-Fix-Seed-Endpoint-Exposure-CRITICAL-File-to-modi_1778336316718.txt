TASK 2: Implement Refresh Token Rotation (CRITICAL)

artifacts/api-server/src/routes/auth.ts (update POST /auth/refresh)
What's broken: Refresh tokens are issued but never rotated. If a refresh token is stolen, an attacker can reuse it forever. This violates OAuth 2.0 best practices and is HIGH RISK.

What to do:

Create a new service tokenRotation.ts with function rotateRefreshToken(oldToken: string)

This function should: a. Verify old token is valid and not revoked b. Decode JWT to get user_id and jti c. Mark old token as revoked in DB (UPDATE refresh_tokens SET revoked = true WHERE jti = $jti) d. Issue NEW access token AND new refresh token e. Return { accessToken, refreshToken, expiresIn } f. Log the rotation for audit

Update POST /auth/refresh route to call this function

Also implement token family tracking:

Add column token_family_id (UUID) to refresh_tokens table
All tokens in the same family share one token_family_id
If a refresh token is used out-of-order, invalidate the ENTIRE family (security signal)

// services/auth/tokenRotation.ts
export async function rotateRefreshToken(oldTokenJWT: string) {
  try {
    // 1. Verify and decode old token
    const decoded = jwt.verify(oldTokenJWT, process.env.JWT_REFRESH_SECRET) as JWTPayload;
    
    // 2. Check token in DB
    const oldToken = await db.select().from(refreshTokens)
      .where(eq(refreshTokens.jti, decoded.jti))
      .limit(1);
    
    if (!oldToken.length || oldToken[0].revoked) {
      throw new Error('Token already revoked or not found');
    }
    
    // 3. Check for suspicious reuse (token family invalidation)
    if (oldToken[0].usedAt && oldToken[0].usedAt > new Date(Date.now() - 10000)) {
      // Token was used very recently — suspicious!
      await db.update(refreshTokens)
        .set({ revokedReason: 'SUSPICIOUS_FAMILY_REUSE' })
        .where(eq(refreshTokens.tokenFamilyId, oldToken[0].tokenFamilyId));
      
      logger.error('[SECURITY] Token family invalidated', { userId: decoded.sub });
      throw new Error('Suspicious token reuse detected. Family invalidated.');
    }
    
    // 4. Revoke old token
    await db.update(refreshTokens)
      .set({ revoked: true, revokedAt: new Date(), revokedReason: 'ROTATED' })
      .where(eq(refreshTokens.jti, decoded.jti));
    
    // 5. Issue new tokens
    const newAccessToken = generateAccessToken(decoded.sub);
    const newRefreshToken = await generateRefreshToken(decoded.sub, oldToken[0].tokenFamilyId);
    
    // 6. Log audit
    logger.info('[AUDIT:AUTH] Refresh token rotated', { userId: decoded.sub });
    
    return { accessToken: newAccessToken, refreshToken: newRefreshToken };
  } catch (err) {
    logger.error('[SECURITY] Token rotation failed', err);
    throw err;
  }
}
